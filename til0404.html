변수

var, let, const

      
        // 여러가지 변수 선언자.
        var a = 1; 
        let b = 2; 
        const c = 3;

        // var와 let은 재할당이 가능하다.
        a = 4;
        b = 5;

        // const는 재할당이 불가능하다.
        c = 6; // error

함수 정의 방법

      
        // 함수 선언문
        function add(a, b) {
          return a + b;
        }
        // 함수 표현식
        const sub = function (a, b) {
          return a - b;
        };
        // 화살표 함수
        const mul = (a, b) => a * b;
        // 즉시 실행 함수
        (function () {
          console.log('즉시 실행 함수');
        })();
